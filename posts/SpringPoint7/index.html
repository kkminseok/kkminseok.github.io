        Appconfig appconfig = new Appconfig();

        //1. мЎ°нҡҢ : к°қмІҙ н•ҳлӮҳ мғқм„ұ

        //2. мЎ°нҡҢ : к°қмІҙ н•ҳлӮҳ мғқм„ұ

    }
}

ліҙмӢңл©ҙмқҖ л‘җ к°қмІҙмқҳ м°ёмЎ°к°Җ лӢ¬лқјм„ң м„ңл№„мҠӨ мҡ”мІӯмӢң л§ӨлІҲ мғқкё°лҠ” кІғмқ„ м•Ң мҲҳ мһҲмҠөлӢҲлӢӨ.

л”°лқјм„ң к°қмІҙк°Җ н•ҳлӮҳл§Ң мғқм„ұлҗҳкі  к·ё к°қмІҙлҘј кіөмң н•ҳлҠ” л°©мӢқмңјлЎң м„Өкі„н•ҳл©ҙ лҗ©лӢҲлӢӨ.

2. мӢұкёҖнҶӨ нҢЁн„ҙ

    • private мғқм„ұмһҗлҘј мӮ¬мҡ©н•ҙ мғҲлЎңмҡҙ к°қмІҙк°Җ мғқкё°лҠ” кІғмқ„ л§үмҠөлӢҲлӢӨ.

package hello.core.singletone;

public class SingletonService {
    //1. static мҳҒм—ӯм—җ к°қмІҙлҘј л”ұ 1к°ңл§Ң мғқм„ұн•©лӢҲлӢӨ.
    private static final SingletonService instance = new SingletonService();

    //2. publicмңјлЎң м—ҙм–ҙм„ң к°қмІҙ мқёмҠӨн„ҙмҠӨк°Җ н•„мҡ”н•ҳл©ҙ getInstance()лҘј нҶөн•ҙ мЎ°нҡҢн•ҳлҸ„лЎқ н—Ҳмҡ©н•©лӢҲлӢӨ.
    public static SingletonService getInstance(){
        return instance;
    }

    //3. private мғқм„ұмһҗлҘј нҶөн•ҙ мғҲлЎңмҡҙ к°қмІҙк°Җ мғқкё°лҠ” кІғмқ„ л§үмҠөлӢҲлӢӨ.
    private SingletonService(){}

    public void logic(){
        System.out.println("мӢұкёҖнҶӨ лЎңм§Ғ к°қмІҙ м¶ңл Ҙ");
    }
}

н•ҳм§Җл§Ң мқҙлҹ¬н•ң мӢұкёҖнҶӨ нҢЁн„ҙм—җлҸ„ лӢЁм җмқҙ мһҲмҠөлӢҲлӢӨ.

  • мӢұкёҖнҶӨ нҢЁн„ҙмқ„ кө¬нҳ„н•ҳлҠ” мҪ”л“ң мһҗмІҙк°Җ л§Һмқҙ л“Өм–ҙк°„лӢӨ.
  • мқҳмЎҙкҙҖкі„мғҒ нҒҙлқјмқҙм–ёнҠёк°Җ кө¬мІҙ нҒҙлһҳмҠӨм—җ мқҳмЎҙн•ңлӢӨ. DIPлҘј мң„л°ҳн•ңлӢӨ.
  • нҒҙлқјмқҙм–ёнҠёк°Җ кө¬мІҙ нҒҙлһҳмҠӨм—җ мқҳмЎҙн•ҙм„ң OCP мӣҗм№ҷмқ„ мң„л°ҳн•  к°ҖлҠҘм„ұмқҙ лҶ’лӢӨ.
  • н…ҢмҠӨнҠён•ҳкё° м–ҙл өлӢӨ.
  • лӮҙл¶Җ мҶҚм„ұмқ„ ліҖкІҪн•ҳкұ°лӮҳ мҙҲкё°нҷ” н•ҳкё° м–ҙл өлӢӨ.
  • private мғқм„ұмһҗлЎң мһҗмӢқ нҒҙлһҳмҠӨлҘј л§Ңл“Өкё° м–ҙл өлӢӨ.
  • кІ°лЎ м ҒмңјлЎң мң м—°м„ұмқҙ л–Ём–ҙм§„лӢӨ.
  • м•ҲнӢ°нҢЁн„ҙмңјлЎң л¶ҲлҰ¬кё°лҸ„ н•ңлӢӨ.

мҠӨн”„л§ҒмқҖ мң„мқҳ лӢЁм җл“Өмқ„ н•ҙкІ°н•ң кё°мҲ мқ„ м ңкіөн•ҳкі  мһҲмҠөлӢҲлӢӨ.

3. мӢұкёҖнҶӨ м»Ён…Ңмқҙл„Ҳ

мҠӨн”„л§Ғ м»Ён…Ңмқҙл„ҲлҠ” мӢұкёҖнҶӨ нҢЁн„ҙмқҳ л¬ём ңм җмқ„ н•ҙкІ°н•ҳл©ҙм„ң, к°қмІҙ мқёмҠӨн„ҙмҠӨлҘј мӢұкёҖнҶӨ(1к°ңл§Ң мғқм„ұ)мңјлЎң кҙҖлҰ¬н•©лӢҲлӢӨ.

м Җнқ¬к°Җ мғқм„ұн–ҲлҚҳ л№Ҳл“Өмқҙ м „л¶Җ мӢұкёҖнҶӨ нҢЁн„ҙмқҙ м Ғмҡ©лҗҳм–ҙ м ҖмһҘлҗҳлҠ” кІғл“Ө мһ…лӢҲлӢӨ.

мҠӨн”„л§Ғ м»Ён…Ңмқҙл„ҲлҠ” мӢұкёҖнҶӨ м»Ён…Ңмқҙл„Ҳ м—ӯн• мқ„ н•©лӢҲлӢӨ. мӢұкёҖнҶӨ к°қмІҙлҘј мғқм„ұн•ҳкі  кҙҖлҰ¬н•ҳлҠ” кё°лҠҘмқ„ мӢұкёҖнҶӨ л Ҳм§ҖмҠӨнҠёлҰ¬лқј н•©лӢҲлӢӨ.

  • лҚ”мқҙмғҒ мҪ”л“ңлҘј м§Җм Җл¶„н•ҳкІҢ мһ‘м„ұм•Ҳн•ҙлҸ„ лҗ©лӢҲлӢӨ.
  • DIP, OCP, н…ҢмҠӨнҠё, private мғқм„ұмһҗлЎң л¶Җн„° мһҗмң лЎӯкІҢ мӢұкёҖнҶӨмқ„ мӮ¬мҡ©н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

н…ҢмҠӨнҠёлҘј мң„н•ҙ мң„м—җ мһ‘м„ұн•ң н…ҢмҠӨнҠё нҢҢмқјм—җ мғҲлЎңмҡҙ н…ҢмҠӨнҠёлҘј л„Јм–ҙмӨҚлӢҲлӢӨ.

м „м—җ л№Ҳм—җ л„Јм–ҙмӨ¬лҚҳ к°қмІҙл“Өмқ„ кәјлӮҙлҙ…лӢҲлӢӨ.

н…ҢмҠӨнҠё кІ°кіјлҘј ліҙл©ҙ к°қмІҙмқҳ м°ёмЎ°к°’мқҙ лҸҷмқјн•ң кІғмқ„ нҷ•мқён•  мҲҳ мһҲмҠөлӢҲлӢӨ.

Note: мҠӨн”„л§Ғмқҳ кё°ліё л№Ҳ л“ұлЎқ л°©мӢқмқҖ мӢұкёҖнҶӨмқҙм§Җл§Ң, мӢұкёҖнҶӨ л°©мӢқл§Ң м§Җмӣҗн•ҳлҠ” кІғмқҖ м•„лӢҲлӢӨ. мҡ”мІӯн•  л•Ң л§ҲлӢӨ мғҲ лЎңмҡҙ к°қмІҙлҘј мғқм„ұн•ҙм„ң л°ҳнҷҳн•ҳлҠ” кё°лҠҘлҸ„ м ңкіөн•©лӢҲлӢӨ. мһҗм„ён•ң лӮҙмҡ©мқҖ л’Өм—җ л№Ҳ мҠӨмҪ”н”„м—җм„ң лӮҳмҳөлӢҲлӢӨ.

н•ҳм§Җл§Ң мқҙлҹ¬н•ң мӢұкёҖнҶӨ л°©мӢқм—җлҸ„ мЈјмқҳн•  м җмқҙ мһҲмҠөлӢҲлӢӨ.

4. мӢұкёҖнҶӨ л°©мӢқмқҳ мЈјмқҳм җ

мӢұкёҖнҶӨ л°©мӢқмқҖ м—¬лҹ¬ нҒҙлқјмқҙм–ёнҠёк°Җ н•ҳлӮҳмқҳ к°қмІҙ мқёмҠӨн„ҙмҠӨлҘј кіөмң н•ҳкё° л•Ңл¬ём—җ мӢұкёҖнҶӨ к°қмІҙлҠ” мғҒнғңлҘј мң м§Җ(stateful)н•ҳкІҢ м„Өкі„н•ҳл©ҙ м•Ҳлҗ©лӢҲлӢӨ.

  • л¬ҙмғҒнғң(stateless)лЎң м„Өкі„н•ҙм•ј н•©лӢҲлӢӨ.
    • нҠ№м • нҒҙлқјмқҙм–ёнҠём—җ мқҳмЎҙм Ғмқё н•„л“ңк°Җ мһҲмңјл©ҙ м•ҲлҗңлӢӨ.
    • нҠ№м • нҒҙлқјмқҙм–ёнҠёк°Җ к°’мқ„ ліҖкІҪн•  мҲҳ мһҲлҠ” н•„л“ңк°Җ мһҲмңјл©ҙ м•ҲлҗңлӢӨ!
    • к°Җкёүм Ғ мқҪкё°л§Ң к°ҖлҠҘн•ҙм•ј н•ңлӢӨ.
    • н•„л“ң лҢҖмӢ м—җ мһҗл°”м—җм„ң кіөмң лҗҳм§Җ м•ҠлҠ”, м§Җм—ӯліҖмҲҳ, нҢҢлқјлҜён„°, ThreadLocal л“ұмқ„ мӮ¬мҡ©н•ҙм•ј н•ңлӢӨ.

л¬ём ңм җмқ„ ліҙкё°мң„н•ҙ mainм—җ вҖҳStatefulServiceвҖҷ мһҗл°” нҢҢмқјмқ„ мғқм„ұн•©лӢҲлӢӨ.

package hello.core;

import org.assertj.core.api.Assertions;
import org.junit.jupiter.api.DisplayName;
import org.junit.jupiter.api.Test;
import org.springframework.context.ApplicationContext;
import org.springframework.context.annotation.AnnotationConfigApplicationContext;
import org.springframework.context.annotation.Bean;

import static org.junit.jupiter.api.Assertions.*;

class StatefulServiceTest {



    @Test
    void StatefulServiceSingleton(){
        ApplicationContext ac = new AnnotationConfigApplicationContext(Testconfig.class);

        //к°қмІҙ л‘җк°ң мғқм„ұ
        StatefulService statefulService1 = ac.getBean("statefulService", StatefulService.class);
        StatefulService statefulService2 = ac.getBean("statefulService", StatefulService.class);

        //userAк°Җ л§Ңмӣҗ мЈјл¬ё
        statefulService1.order("userA",10000);

        //userBк°Җ 2л§Ңмӣҗ мЈјл¬ё
        statefulService2.order("userB",20000);

        //мЎ°нҡҢ л§Ңмӣҗмқҙм°ҚнҳҖм•јн•Ё.
        int price = statefulService1.getPrice();

        //2л§Ңмӣҗ м°Қнһҳ.

        Assertions.assertThat(price).isEqualTo(20000);
    }



    static class Testconfig{
        @Bean
        public StatefulService statefulService(){
            return new StatefulService();
        }
    }

}
