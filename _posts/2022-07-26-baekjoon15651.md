## 5. code

```python
n, m = map(int, input().split())


def solution(li, m):
    size = len(li)
    pick = []

    def recur():
        if (len(pick) == m):
            for idx in pick:
                print(idx, end=" ")
            print("\n", end="")
            return

        for i in range(1, size + 1):
            pick.append(i)
            recur()
            pick.pop()

    recur()


li = list(range(1, n + 1))

solution(li, m)

```
